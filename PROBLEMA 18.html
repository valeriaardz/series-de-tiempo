# Import necessary libraries
import pandas as pd
import numpy as np
from google.colab import files  # For file upload in Colab
import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller  # For stationarity test
from statsmodels.tsa.arima.model import ARIMA  # For ARIMA modeling
import warnings
warnings.filterwarnings("ignore")  # Suppress warnings for cleaner output

# Step 1: Upload the Excel file containing PDI data
print("Please upload the Excel file containing the PDI data")
uploaded = files.upload()
filename = list(uploaded.keys())[0]  # Get the uploaded file's name
df = pd.read_excel(filename)  # Read the Excel file into a DataFrame

# Step 2: Display available columns and prompt user to select the PDI column
print("\nAvailable columns in the dataset:")
print(df.columns.tolist())
column_name = input("Please enter the column name containing Personal Disposable Income (e.g., 'PDI'): ")

# Verify the selected column exists
if column_name not in df.columns:
    raise ValueError(f"Column '{column_name}' not found in the dataset. Please check the column name.")

# Step 3: Prepare the data - Take the log of PDI and handle any missing values
# Assuming PDI data is numeric; convert to numeric if needed and drop NaNs
df[column_name] = pd.to_numeric(df[column_name], errors='coerce')
df = df.dropna(subset=[column_name])  # Drop rows where PDI is missing
log_pdi = np.log(df[column_name])  # Compute the natural log of PDI

# Step 4: Check for stationarity using Augmented Dickey-Fuller (ADF) test
def adf_test(series):
    result = adfuller(series)
    print("\nADF Test Results:")
    print(f"ADF Statistic: {result[0]}")
    print(f"p-value: {result[1]}")
    print("Critical Values:")
    for key, value in result[4].items():
        print(f"   {key}: {value}")
    if result[1] < 0.05:
        print("Result: Stationary (reject null hypothesis)")
    else:
        print("Result: Non-stationary (fail to reject null hypothesis)")

# Perform ADF test on log PDI
print("\nTesting stationarity of log(PDI):")
adf_test(log_pdi)

# Step 5: Difference the data if non-stationary (common for economic data like PDI)
log_pdi_diff = log_pdi.diff().dropna()  # First difference and remove NaN
print("\nTesting stationarity of differenced log(PDI):")
adf_test(log_pdi_diff)

# Step 6: Fit an ARIMA model
# For this example, we'll try ARIMA(1,1,1) based on differencing (d=1)
# In practice, you'd use ACF/PACF plots or auto_arima for order selection
print("\nFitting ARIMA(1,1,1) model to log(PDI):")
model = ARIMA(log_pdi, order=(1, 1, 1))  # p=1, d=1, q=1
model_fit = model.fit()

# Step 7: Display model summary
print("\nARIMA Model Summary:")
print(model_fit.summary())

# Step 8: Plot diagnostics to check residuals
model_fit.plot_diagnostics(figsize=(12, 8))
plt.show()

# Step 9: Make predictions (e.g., next 5 periods)
forecast_steps = 5
forecast = model_fit.forecast(steps=forecast_steps)
forecast_index = pd.date_range(start=df.index[-1], periods=forecast_steps + 1, freq='Q')[1:]  # Quarterly data assumption
forecast_series = pd.Series(forecast, index=forecast_index)

# Step 10: Plot original data and forecast
plt.figure(figsize=(10, 6))
plt.plot(log_pdi, label='Log PDI (Observed)')
plt.plot(forecast_series, label='Forecast', color='red')
plt.title('Log Personal Disposable Income with ARIMA Forecast')
plt.xlabel('Time')
plt.ylabel('Log PDI')
plt.legend()
plt.grid(True)
plt.show()

# Optional: Convert forecast back to original scale (exponentiate)
forecast_original_scale = np.exp(forecast)
print(f"\nForecasted PDI (original scale) for next {forecast_steps} periods:")
for date, value in zip(forecast_index, forecast_original_scale):
    print(f"{date}: {value:.2f}")
